40  static bool intersects(bounds_2d<Ta> a, bounds_2d<Ta> b) {
41  return (
42  a.right >= b.left &&
43  a.left <= b.right &&
44  a.top >= b.bottom &&
45  a.bottom <= b.top);
46  }
47  static bool within(bounds_2d<Ta> a, bounds_2d<Ta> b) {
48  return (
49  a.left >= b.left &&
50  a.right <= b.right &&
51  a.top <= b.top &&
52  a.bottom >= b.bottom);
53  }
54  static bool outside(bounds_2d<Ta> a, bounds_2d<Ta> b) {
55  return !intersects(a, b);

58  bounds_2d<Ta> out;

60  out.right = fmax(a.right, b.right);

75  bounds_2d<Ta> transform(std::string srs_from, std::string srs_to) {
76  if (srs_from == srs_to) {
77  return *this;
78  }
79  OGRSpatialReference srs_in;
80  OGRSpatialReference srs_out;
81  srs_in.SetFromUserInput(srs_from.c_str());
82  srs_out.SetFromUserInput(srs_to.c_str());
83 
84  if (srs_in.IsSame(&srs_out)) {
85  return *this;
86  }
87 
88  OGRCoordinateTransformation* coord_transform = OGRCreateCoordinateTransformation(&srs_in, &srs_out);
89 
90  Ta x[4] = {left, left, right, right};
91  Ta y[4] = {top, bottom, top, bottom};
92 
93  if (coord_transform == NULL || !coord_transform->Transform(4, x, y)) {
94  throw std::string("ERROR: coordinate transformation failed (from " + srs_from + " to " + srs_to + ").");
95  }
96 
97  Ta xmin = std::numeric_limits<Ta>::is_integer ? std::numeric_limits<Ta>::max() : std::numeric_limits<Ta>::max();
98  Ta ymin = std::numeric_limits<Ta>::is_integer ? std::numeric_limits<Ta>::max() : std::numeric_limits<Ta>::max();
99  Ta xmax = std::numeric_limits<Ta>::is_integer ? std::numeric_limits<Ta>::min() : -std::numeric_limits<Ta>::max();
100  Ta ymax = std::numeric_limits<Ta>::is_integer ? std::numeric_limits<Ta>::min() : -std::numeric_limits<Ta>::max();
101  for (uint8_t k = 0; k < 4; ++k) {
102  if (x[k] < xmin) xmin = x[k];
103  if (y[k] < ymin) ymin = y[k];
104  if (x[k] > xmax) xmax = x[k];
105  if (y[k] > ymax) ymax = y[k];
106  }
107 
108  left = xmin;
109  right = xmax;
110  top = ymax;
111  bottom = ymin;
112 
113  OCTDestroyCoordinateTransformation(coord_transform);
114 
115  return *this;
116  }
117 };
