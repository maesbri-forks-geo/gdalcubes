62  static std::shared_ptr<join_bands_cube> create(std::vector<std::shared_ptr<cube>> in_cubes, std::vector<std::string> prefixes = {}) {
63  std::shared_ptr<join_bands_cube> out = std::make_shared<join_bands_cube>(in_cubes, prefixes);

72  join_bands_cube(std::vector<std::shared_ptr<cube>> in_cubes, std::vector<std::string> prefixes = {}) : cube(), _in(in_cubes), _prefix(prefixes) {
73  _st_ref = std::make_shared<cube_stref_regular>();
74 
75  if (_in.size() < 2) {
76  throw std::string("ERROR in join_bands_cube::join_bands_cube(): Expected at least two input data cubes");
77  }
78 
79  if (!_prefix.empty() && (in_cubes.size() != prefixes.size())) {
80  throw std::string("ERROR in join_bands_cube::join_bands_cube(): The number of name prefixes does not match the number of provided input data cubes");
81  }
82 
83  for (uint16_t i = 1; i < in_cubes.size(); ++i) {
84  if (cube_stref::type_string(_in[0]->st_reference()) != cube_stref::type_string(_in[i]->st_reference())) {
85  throw std::string("ERROR in join_bands_cube::join_bands_cube(): Incompatible spatial / temporal reference types");
86  }
87 
88  if (cube_stref::type_string(_in[0]->st_reference()) == "cube_stref_regular") {
89  std::shared_ptr<cube_stref_regular> stref_A = std::dynamic_pointer_cast<cube_stref_regular>(_in[0]->st_reference());
90  std::shared_ptr<cube_stref_regular> stref_B = std::dynamic_pointer_cast<cube_stref_regular>(_in[i]->st_reference());
91  // Check that A and B have identical shape
92  if (*(stref_A) != *(stref_B)) {
93  throw std::string("ERROR in join_bands_cube::join_bands_cube(): Cubes have different shape");
94  }
95  } else if (cube_stref::type_string(_in[0]->st_reference()) == "cube_stref_labeled_time") {
96  std::shared_ptr<cube_stref_labeled_time> stref_A = std::dynamic_pointer_cast<cube_stref_labeled_time>(_in[0]->st_reference());
97  std::shared_ptr<cube_stref_labeled_time> stref_B = std::dynamic_pointer_cast<cube_stref_labeled_time>(_in[i]->st_reference());
98  // Check that A and B have identical shape
99  if (*(stref_A) != *(stref_B)) {
100  throw std::string("ERROR in join_bands_cube::join_bands_cube(): Cubes have different shape");
101  }
102  }
103 
104  if (!(_in[0]->chunk_size()[0] == _in[i]->chunk_size()[0] &&
105  _in[0]->chunk_size()[1] == _in[i]->chunk_size()[1] &&
106  _in[0]->chunk_size()[2] == _in[i]->chunk_size()[2])) {
107  throw std::string("ERROR in join_bands_cube::join_bands_cube(): Cubes have different chunk sizes");
108  }
109 
110  if (_prefix.empty()) {
111  bool has_name_conflicts = false;
112  // check that there are no name conflicts
113  for (uint16_t iba = 0; iba < _in[0]->bands().count(); ++iba) {
114  if (_in[i]->bands().has(_in[i]->bands().get(iba).name)) {
115  has_name_conflicts = true;
116  break;
117  }
118  }
119  if (has_name_conflicts) {
120  GCBS_WARN("Input cubes have bands with identical names, default name prefixes will be added.");
121  for (uint16_t j = 0; j < in_cubes.size(); ++j) {
122  _prefix.push_back("X" + std::to_string(j + 1));
123  }
124  }
125  } else {
126  if (_prefix[0] == _prefix[i]) {
127  GCBS_ERROR("cannot join cubes with identical prefixes");
128  throw std::string("ERROR in join_bands_cube::join_bands_cube(): Cannot join cubes with identical prefixes");
129  }
130  }
131  }
132 
133  _st_ref = _in[0]->st_reference();
134 
135  _chunk_size[0] = _in[0]->chunk_size()[0];
136  _chunk_size[1] = _in[0]->chunk_size()[1];
137  _chunk_size[2] = _in[0]->chunk_size()[2];
138 
139  for (uint16_t i = 0; i < in_cubes.size(); ++i) {
140  for (uint16_t ib = 0; ib < _in[i]->bands().count(); ++ib) {
141  band b = _in[i]->bands().get(ib);
142  b.name = _prefix.empty() ? (b.name) : (_prefix[i] + "." + b.name);
143  _bands.add(b);
144  }
145  }
146  }
